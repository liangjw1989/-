            function getRatingForConcentration(conc) {
                if (conc > STANDARDS.excellent.conc) {
                    return { score: STANDARDS.excellent.score, rating: STANDARDS.excellent.rating,
                        tagClass: STANDARDS.excellent.tagClass, level: 'excellent' };
                } else if (conc > STANDARDS.good.conc) {
                    return { score: STANDARDS.good.score, rating: STANDARDS.good.rating, tagClass: STANDARDS.good
                            .tagClass, level: 'good' };
                } else if (conc > STANDARDS.pass.conc) {
                    return { score: STANDARDS.pass.score, rating: STANDARDS.pass.rating, tagClass: STANDARDS.pass
                            .tagClass, level: 'pass' };
                } else {
                    return { score: STANDARDS.fail.score, rating: STANDARDS.fail.rating, tagClass: STANDARDS.fail
                            .tagClass, level: 'fail' };
                }
            }

            function getRatingForPR(pr) {
                if (pr > STANDARDS.excellent.pr) {
                    return { score: STANDARDS.excellent.score, rating: STANDARDS.excellent.rating,
                        tagClass: STANDARDS.excellent.tagClass, level: 'excellent' };
                } else if (pr > STANDARDS.good.pr) {
                    return { score: STANDARDS.good.score, rating: STANDARDS.good.rating, tagClass: STANDARDS.good
                            .tagClass, level: 'good' };
                } else if (pr > STANDARDS.pass.pr) {
                    return { score: STANDARDS.pass.score, rating: STANDARDS.pass.rating, tagClass: STANDARDS.pass

            function getOverallRating(concLevel, prLevel) {

                const prIndex = levelOrder.indexOf(prLevel);

                const lowerLevel = levelOrder[lowerIndex];
                const stdKey = lowerLevel === 'fail' ? 'fail' : lowerLevel;
                return {
                    rating: STANDARDS[stdKey].rating,
                    score: STANDARDS[stdKey].score,
                    tagClass: STANDARDS[stdKey].tagClass,
                    level: lowerLevel,
                    pregnancyDesc: STANDARDS[stdKey].pregnancyDesc,
                };

            function calculateScore() {

                // 管理员密码检测

                const conc = parseFloat(concRaw);
                const pr = parseFloat(prRaw);

                    return;

                const prResult = getRatingForPR(pr);
                const overall = getOverallRating(concResult.level, prResult.level);

                currentResult = {

                    concScore: concResult.score,

                    prRating: prResult.rating,
